//returning the sum in decimal by using the convert function
int sum(string c) {
	int length = c.length();
	int i, sum = 0;
	int x = 0;
	for (i = length - 1; i >= 0; i--) {
		sum = sum + pow(16, i)*convert(c[x]);
		x++;
	}
	return sum;
}
